open and close

package sample01; import org.openqa.selenium.WebDriver; import org.openqa.selenium.chrome.ChromeDriver; public class openandclose { public static void main(String[] args)throws InterruptedException { System.setProperty("webdriver.chrome.driver","C:\\Program Files (x86)\\chromedriver.exe"); WebDriver driver =new ChromeDriver(); try { driver.get("https://pixabay.com"); System.out.println("Website is opened successfully"); Thread.sleep(5000); } finally { driver.quit(); System.out.println("Website is closed successfully"); } } }
