Open and Close

package sample01;
import org.openqa.selenium.WebDriver;
import org.openqa.selenium.chrome.ChromeDriver;

public class openandclose {
	public static void main(String[] args) throws InterruptedException {
		System.setProperty("webdriver.chrome.driver","C:\\Program Files (x86)\\chromedriver.exe");
		WebDriver driver = new ChromeDriver();
		try {
			driver.get("https://pixabay.com");
			System.out.println("Website is opened successfully");
			Thread.sleep(5000);
		} finally {
			driver.quit();
			System.out.println("Website is closed successfully");
		}
	}
}

Link Text

package sample01;
import org.openqa.selenium.*;
import org.openqa.selenium.chrome.ChromeDriver;

public class linktextpg {

	public static void main(String[] args) throws InterruptedException {
		System.setProperty("webdriver.chrome.driver", "C:\\Program Files (x86)\\chromedriver.exe");
		WebDriver driver = new ChromeDriver();
		driver.get("https://www.leafground.com/link.xhtml");
		WebElement hyperlink = driver.findElement(By.linkText("Go to Dashboard"));
		hyperlink.click();
		try {
			System.out.println("Dashboard is opened successfully");
			Thread.sleep(3000);
			System.out.println("Navigating back to Link page.....");
		} finally {
			driver.navigate().back();
			Thread.sleep(3000);
			WebElement duplicate = driver.findElement(By.partialLinkText("Broken"));
			duplicate.click();
			System.out.println("Broken link tested successfully is closed successfully");
		}
	}
}
